Instantiating

			var options = {	// these are default values
				animationStep: 100,
				chartColor: '#0bbaba',
				cutOut: 98,
				bgColor: '#444444',
				doughnutPadding: 5,
				dotColor: '#0bbaba',
				outerCanvasSize: 250,
				pointSize: 5
			}

			mmtDonutChart('chart', 75, options);	// put ID, percent, options

Demos

Options

			{
				// Number - Amount of animation steps
				animationStep: 100,

				// String - Animation part color of chart
				chartColor: '#0bbaba',

				// Number - The percentage of the chart that we cut out of the middle
				cutOut: 98,

				// String - Whole donut chart background color
				bgColor: '#444444',

				// Number - Padding(px) inside outer canvas 
				doughnutPadding: 5,

				// String - Donut chart dot color
				dotColor: '#0bbaba',

				// Number - Chart outer canvas size(px)
				outerCanvasSize: 250,

				// Number - Size(px) of dot
				pointSize: 5
			}
