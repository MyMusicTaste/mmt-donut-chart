				function progressCounter(labelName, percent){
					var i=0;
					var timer = 500/percent
					var num = document.getElementById(labelName);

					setInterval(function(){
						if(i<=percent){
							num.innerHTML = i;
							i++;
						}
					}, timer);
				}

				mmtDonutChart('chart', 100);  // put chart ID, percent, options
				progressCounter('chart-num', 100);	  // put label ID

Options

			{
				// Number - Amount of animation steps
				animationStep: 100,

				// String - Animation part color of chart
				chartColor: '#0bbaba',

				// Number - The percentage of the chart that we cut out of the middle
				cutOut: 98,

				// String - Whole donut chart background color
				bgColor: '#444444',

				// Number - Padding(px) inside outer canvas 
				doughnutPadding: 5,

				// String - Donut chart dot color
				dotColor: '#0bbaba',

				// Number - Chart outer canvas size(px)
				outerCanvasSize: 230,

				// Number - Size(px) of dot
				pointSize: 5
			}
